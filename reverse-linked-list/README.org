Reverse Linked List
---------------------------------------------------------------------
Reverse a singly linked list.

class Solution(object):
    def reverseList(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        ## Basic Idea:
        ##   1 -> 3 -> 4 -> 5
        ##   p   q   r
        ##   always insert to the head
        if head is None or head.next is None:
            return head

        p = head
        q = p.next
        p.next = None

        while True:
            r = q.next
            q.next = p
            p = q
            q = r
            if q is None:
                break
        return p
