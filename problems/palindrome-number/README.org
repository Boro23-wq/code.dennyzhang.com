## Basic Ideas:  Generate a new integer Y from right digit to left digit
##               Then compare it with X
##         1234321: 1 -> 12 -> 13 -> 134 ...
## Complexity: Time O(1), Space O(1). The integer can be at most 32 digits
class Solution(object):
    def isPalindrome(self, x):
        """
        :type x: int
        :rtype: bool
        """
        if x < 0:
            return False
        if x == 0:

            return True
        val = x
        y = 0
        while val != 0:
            y = 10*y + (val%10)
            val = val / 10
        return x == y
